### 题目描述

将单向链表按某值划分成左边小、 中间相等、 右边大的形式

如果链表长度为N， 时间复杂度请达到O(N)， 额外空间复杂度请达到O(1)

### 思路解析

定义6个辅助指针：left_head, left_tail, mid_head, mid_tail, right_head, right_tail，分别表示左边，中间，右边的头尾指针

遍历链表，拆分组成三个新的链表，最后再合并


```python

class Node:
    def __init__(self, data):
        self.next = None
        self.data = data

def partitionLinkedList(head, pivot):
    left_head, left_tail, mid_head, mid_tail, right_head, right_tail = None, None, None, None, None, None
    # 遍历链表，拆分组成三个新的链表
    while head:
        temp = head.next
        # 将当前节点拆下加入新的链表
        head.next = None
        if head.data < pivot:
            if not left_head:
                left_head = head
                left_tail = head
            else:
                left_tail.next = head
                left_tail = left_tail.next
        elif head.data == pivot:
            if not mid_head:
                mid_head = head
                mid_tail = head
            else:
                mid_tail.next = head
                mid_tail = mid_tail.next
        else:
            if not right_head:
                right_head = head
                right_tail = head
            else:
                right_tail.next = head
                right_tail = right_tail.next

        head = temp

    # 链表组合
    if left_tail:
        left_tail.next = mid_head
        mid_tail = mid_tail if mid_tail else left_tail

    if mid_tail:
        mid_tail.next = right_head

    if left_head:
        return left_head
    elif mid_head:
        return mid_head
    else:
        return right_head


```

```python

class Solution:
    def partition(self, head: ListNode, x: int) -> ListNode:
        leftHead, midHead, rightHead = ListNode(-1), ListNode(-1), ListNode(-1)

        leftTail = leftHead
        midTail = midHead
        rightTail = rightHead
        while head:
            if head.val < x:
                leftTail.next = head
                leftTail = leftTail.next
            elif head.val == x:
                midTail.next = head
                midTail = midTail.next
            else:
                rightTail.next = head
                rightTail = rightTail.next
            head = head.next

        if midHead.next:
            leftTail.next = midHead.next
            midTail.next = rightHead.next
        else:
            leftTail.next = rightHead.next

        return leftHead.next

```
